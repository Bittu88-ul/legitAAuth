Developer Documentation

Welcome to the LegitAuth integration guide. Choose your authentication method below.

Initialization (Required for Both)

Add Auth.cs to your C# Project.

⚠️ Note: If you get a "System.Text.Json" error, please install the System.Text.Json package via NuGet Package Manager.

using Auth; // Add this at the very top!

// In Form Load
private void Form1_Load(object sender, EventArgs e)
{
    LegitAuthApp.init(
        "APP_NAME",
        "OWNER_ID",
        "SECRET",
        "1.0"
    );
    
    // Optional: Auto-load saved credentials
    string savedUser, savedPass;
    LegitAuthApp.LoadCredentials(out savedUser, out savedPass);
    User.Text = savedUser;
    Pass.Text = savedPass;
}

Method 1: Username & Password Login

// In Login Button
private async void btnLogin_Click(object sender, EventArgs e)
{
    await LegitAuthApp.login(User.Text, Pass.Text);
    
    if (LegitAuthApp.response.success)
    {
        // Auto-save credentials for next time
        LegitAuthApp.SaveCredentials(User.Text, Pass.Text);
        
        Form1 main = new Form1();
        main.Show();
        this.Hide();
    }
    else
    {
        status.Text = LegitAuthApp.response.message;
        Console.Beep(400, 300);
    }
}

Method 2: License Key Login

// In Login Button
private async void btnLoginKey_Click(object sender, EventArgs e)
{
    await LegitAuthApp.license(License.Text);
    
    if (LegitAuthApp.response.success)
    {
        Form1 main = new Form1();
        main.Show();
        this.Hide();
    }
    else
    {
        status.Text = LegitAuthApp.response.message;
        Console.Beep(400, 300);
    }
}
