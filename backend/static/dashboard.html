Developer Documentation

Step-by-step guide to integrate the LegitAuth SDK into your C# application.

Download C# SDK Wrapper

Fully async wrapper built on modern System.Text.Json dependencies. Includes automatic local credentials saving utilities.

Download Auth.cs

Step 1: Add Auth.cs to Your Project

First, download the Auth.cs wrapper file using the button above. Copy this file into your Visual Studio project directory, right-click your project in the Solution Explorer, select Add > Existing Item..., and choose Auth.cs.

At the top of your form class file (e.g. LoginForm.cs), add the namespace reference:

using Auth; // Load the LegitAuth namespace

Step 2: Initialize in Form_Load

Double-click your login form in the Visual Studio designer to create the Form_Load event handler. Put the initialization code there. You can copy the credentials from your app card in the dashboard.

private void LoginForm_Load(object sender, EventArgs e)
{
    // 1. Initialize application credentials
    LegitAuthApp.init(
        "APP_NAME",         // Your Application Name
        "OWNER_ID",         // Developer Owner ID
        "SECRET",           // App Shared Secret
        "1.0"               // App Version
    );
    
    // 2. Optional: Load previously saved credentials to autofill inputs
    string savedUser, savedPass;
    LegitAuthApp.LoadCredentials(out savedUser, out savedPass);
    if (!string.IsNullOrEmpty(savedUser))
    {
        txtUsername.Text = savedUser;
        txtPassword.Text = savedPass;
        chkRememberMe.Checked = true; // Set check if you have a remember me checkbox
    }
}

Step 3: Implement User/Password Login Button

Double-click your Login button to create the click handler. Because our API runs asynchronously to keep the UI responsive, make sure to add the async keyword to your event handler method name.

private async void btnLogin_Click(object sender, EventArgs e)
{
    string username = txtUsername.Text.Trim();
    string password = txtPassword.Text;

    if (string.IsNullOrEmpty(username) || string.IsNullOrEmpty(password))
    {
        lblStatus.Text = "Please enter username and password.";
        return;
    }

    // Trigger async login request
    await LegitAuthApp.login(username, password);

    if (LegitAuthApp.response.success)
    {
        // Optional: Save user credentials locally if "Remember Me" is checked
        if (chkRememberMe.Checked)
        {
            LegitAuthApp.SaveCredentials(username, password);
        }
        else
        {
            LegitAuthApp.SaveCredentials("", ""); // Clear credentials cache
        }

        MessageBox.Show("Welcome back!", "Success", MessageBoxButtons.OK, MessageBoxIcon.Information);
        
        // Load main window and hide login window
        MainForm dashboard = new MainForm();
        dashboard.Show();
        this.Hide();
    }
    else
    {
        // Show server response error (e.g. invalid password, account banned, HWID mismatch)
        lblStatus.Text = LegitAuthApp.response.message;
        Console.Beep(400, 300); // Alert audio warning
    }
}

Step 4: Implement License Key Login Button

If you prefer single-use license key authentication instead of username/password, use the license method. Implement this click handler for your key entry button:

private async void btnLoginKey_Click(object sender, EventArgs e)
{
    string key = txtLicenseKey.Text.Trim();

    if (string.IsNullOrEmpty(key))
    {
        lblStatus.Text = "Please enter your license key.";
        return;
    }

    // Trigger async license verification
    await LegitAuthApp.license(key);

    if (LegitAuthApp.response.success)
    {
        MessageBox.Show("License verified successfully!", "Welcome", MessageBoxButtons.OK, MessageBoxIcon.Information);
        
        MainForm dashboard = new MainForm();
        dashboard.Show();
        this.Hide();
    }
    else
    {
        // Show failure status (e.g. key expired, banned, invalid key)
        lblStatus.Text = LegitAuthApp.response.message;
        Console.Beep(400, 300);
    }
}
